## 1. 执行摘要

### 1.1 项目定位

DayPilot是一个基于纯原生技术的日程管理Web应用，采用单HTML文件架构，具备日/周/月多视图切换、任务分类管理、项目集管理、数据导出等核心功能。

### 1.2 核心发现

| 维度 | 现状评分 | 主要问题 |
|------|----------|----------|
| 代码架构 | ⭐⭐⭐ | 单文件耦合、全局变量污染 |
| 性能表现 | ⭐⭐⭐⭐ | 整体良好，大数据量下有优化空间 |
| 用户体验 | ⭐⭐⭐ | 缺少搜索、拖拽调整、完成状态 |
| 功能完整度 | ⭐⭐⭐ | 基础功能齐全，高级功能缺失 |
| 可维护性 | ⭐⭐ | 无模块化、注释不足 |

### 1.3 建议优先级

```
紧急且重要：
├── 代码模块化重构
├── 添加任务搜索功能
└── 实现任务完成状态

重要不紧急：
├── 数据可视化统计
├── 拖拽调整任务时间
└── 重复任务支持

优化类：
├── 性能优化（虚拟滚动）
├── 移动端适配增强
└── 主题切换（暗色模式）
```

**重构示例 - 状态管理模块：**

```javascript
// js/store/index.js
class Store {
  constructor() {
    this.state = {
      tasks: [],
      categories: [],
      projects: [],
      currentView: 'week',
      currentDate: new Date(),
      filterCatId: null,
      filterProjId: null,
      editingId: null
    };
    this.listeners = new Map();
  }

  // 获取状态
  getState() {
    return { ...this.state };
  }

  // 更新状态
  setState(updates) {
    this.state = { ...this.state, ...updates };
    this.notify(updates);
  }

  // 订阅状态变化
  subscribe(key, callback) {
    if (!this.listeners.has(key)) {
      this.listeners.set(key, new Set());
    }
    this.listeners.get(key).add(callback);
    return () => this.listeners.get(key).delete(callback);
  }

  // 通知订阅者
  notify(updates) {
    Object.keys(updates).forEach(key => {
      if (this.listeners.has(key)) {
        this.listeners.get(key).forEach(cb => cb(this.state[key]));
      }
    });
  }
}

export const store = new Store();
```

**重构示例 - 任务模块：**

```javascript
// js/store/tasks.js
import { store } from './index.js';
import { storage } from '../utils/storage.js';

class TaskStore {
  constructor() {
    this.storageKey = 'sched_tasks';
  }

  // 加载任务
  async load() {
    const tasks = await storage.get(this.storageKey) || [];
    store.setState({ tasks });
  }

  // 保存任务
  async save() {
    const { tasks } = store.getState();
    await storage.set(this.storageKey, tasks);
  }

  // 添加任务
  async add(taskData) {
    const { tasks } = store.getState();
    const newTask = {
      id: this.generateId(),
      ...taskData,
      createdAt: Date.now(),
      completed: false
    };
    store.setState({ tasks: [...tasks, newTask] });
    await this.save();
    return newTask;
  }

  // 更新任务
  async update(id, updates) {
    const { tasks } = store.getState();
    const index = tasks.findIndex(t => t.id === id);
    if (index === -1) return null;

    const updatedTask = { ...tasks[index], ...updates };
    const newTasks = [...tasks];
    newTasks[index] = updatedTask;
    store.setState({ tasks: newTasks });
    await this.save();
    return updatedTask;
  }

  // 删除任务
  async delete(id) {
    const { tasks } = store.getState();
    store.setState({ tasks: tasks.filter(t => t.id !== id) });
    await this.save();
  }

  // 搜索任务
  search(keyword) {
    const { tasks } = store.getState();
    const lowerKeyword = keyword.toLowerCase();
    return tasks.filter(t => 
      t.title.toLowerCase().includes(lowerKeyword) ||
      (t.project && t.project.toLowerCase().includes(lowerKeyword))
    );
  }

  // 生成唯一ID
  generateId() {
    return 'task_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 7);
  }
}

export const taskStore = new TaskStore();
```

### 3.2 状态管理优化

**当前问题：**
- 状态分散在30+个全局变量中
- 状态变更难以追踪
- 组件间数据传递混乱

**优化方案 - 响应式状态管理：**

```javascript
// 简单的响应式系统
class ReactiveState {
  constructor(initialState) {
    this._state = initialState;
    this._deps = new Map();
    this._watchers = new Map();
  }

  get(key) {
    // 收集依赖
    if (this._currentEffect) {
      if (!this._deps.has(key)) {
        this._deps.set(key, new Set());
      }
      this._deps.get(key).add(this._currentEffect);
    }
    return this._state[key];
  }

  set(key, value) {
    const oldValue = this._state[key];
    if (oldValue === value) return;
    
    this._state[key] = value;
    
    // 触发更新
    if (this._deps.has(key)) {
      this._deps.get(key).forEach(effect => effect(value, oldValue));
    }
    
    // 触发watcher
    if (this._watchers.has(key)) {
      this._watchers.get(key).forEach(cb => cb(value, oldValue));
    }
  }

  watch(key, callback) {
    if (!this._watchers.has(key)) {
      this._watchers.set(key, new Set());
    }
    this._watchers.get(key).add(callback);
    return () => this._watchers.get(key).delete(callback);
  }
}

// 使用示例
const state = new ReactiveState({
  tasks: [],
  currentView: 'week',
  currentDate: new Date()
});

// 监听变化
state.watch('tasks', (newTasks) => {
  renderTaskList(newTasks);
  updateStatistics(newTasks);
});

state.watch('currentView', (view) => {
  switchView(view);
});
```

```javascript
// 简化的虚拟DOM实现
class VNode {
  constructor(tag, props, children) {
    this.tag = tag;
    this.props = props || {};
    this.children = children || [];
    this.el = null;
  }
}

function createElement(vnode) {
  if (typeof vnode === 'string') {
    return document.createTextNode(vnode);
  }
  
  const el = document.createElement(vnode.tag);
  
  // 设置属性
  Object.entries(vnode.props).forEach(([key, value]) => {
    if (key.startsWith('on')) {
      el.addEventListener(key.slice(2).toLowerCase(), value);
    } else if (key === 'className') {
      el.className = value;
    } else if (key === 'style' && typeof value === 'object') {
      Object.assign(el.style, value);
    } else {
      el.setAttribute(key, value);
    }
  });
  
  // 递归创建子元素
  vnode.children.forEach(child => {
    el.appendChild(createElement(child));
  });
  
  vnode.el = el;
  return el;
}

function diff(oldVNode, newVNode) {
  // 简化的diff逻辑
  if (!oldVNode) return createElement(newVNode);
  if (!newVNode) return null;
  if (oldVNode.tag !== newVNode.tag) return createElement(newVNode);
  
  // 更新属性
  const el = oldVNode.el;
  updateProps(el, oldVNode.props, newVNode.props);
  
  // 更新子节点
  diffChildren(el, oldVNode.children, newVNode.children);
  
  newVNode.el = el;
  return el;
}
```

```javascript
// 优化：虚拟滚动
class VirtualScroller {
  constructor(container, itemHeight, renderItem) {
    this.container = container;
    this.itemHeight = itemHeight;
    this.renderItem = renderItem;
    this.items = [];
    this.visibleItems = [];
    
    this.viewport = document.createElement('div');
    this.viewport.style.height = '100%';
    this.viewport.style.overflow = 'auto';
    this.container.appendChild(this.viewport);
    
    this.content = document.createElement('div');
    this.content.style.position = 'relative';
    this.viewport.appendChild(this.content);
    
    this.viewport.addEventListener('scroll', () => this.onScroll());
  }
  
  setItems(items) {
    this.items = items;
    this.content.style.height = (items.length * this.itemHeight) + 'px';
    this.render();
  }
  
  onScroll() {
    requestAnimationFrame(() => this.render());
  }
  
  render() {
    const scrollTop = this.viewport.scrollTop;
    const viewportHeight = this.viewport.clientHeight;
    
    const startIndex = Math.floor(scrollTop / this.itemHeight);
    const endIndex = Math.min(
      startIndex + Math.ceil(viewportHeight / this.itemHeight) + 1,
      this.items.length
    );
    
    // 只渲染可见项
    this.content.innerHTML = '';
    for (let i = startIndex; i < endIndex; i++) {
      const el = this.renderItem(this.items[i], i);
      el.style.position = 'absolute';
      el.style.top = (i * this.itemHeight) + 'px';
      el.style.width = '100%';
      this.content.appendChild(el);
    }
  }
}
```

```javascript
// 优化：使用防抖和节流
function debounce(fn, delay) {
  let timer = null;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

function throttle(fn, limit) {
  let inThrottle = false;
  return function(...args) {
    if (!inThrottle) {
      fn.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}
```

```javascript
// 统一错误处理
class ErrorHandler {
  static async wrap(promise, errorMessage) {
    try {
      return await promise;
    } catch (error) {
      console.error(errorMessage, error);
      ErrorHandler.showUserError(errorMessage);
      throw error;
    }
  }
  
  static showUserError(message, duration = 3000) {
    const toast = document.createElement('div');
    toast.className = 'error-toast';
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.classList.add('fade-out');
      setTimeout(() => toast.remove(), 300);
    }, duration);
  }
}
```

### 3.5 数据存储方案对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| localStorage | 简单、兼容性好 | 容量限制(5MB)、同步操作 | 小数据量、简单应用 |
| IndexedDB | 容量大、异步、支持索引 | API复杂、学习成本高 | 大数据量、复杂查询 |
| File System API | 直接操作文件、数据可移植 | 兼容性差、需要用户授权 | 数据导出、文件协作 |
| OPFS | 高性能、沙箱环境 | 新API、兼容性有限 | 高性能文件操作 |

**建议方案：混合存储策略**

```javascript
class HybridStorage {
  constructor() {
    this.primary = new IndexedDBStorage();
    this.fallback = new LocalStorageAdapter();
    this.file = new FileSystemStorage();
  }
  
  async get(key) {
    try {
      // 优先使用IndexedDB
      return await this.primary.get(key);
    } catch {
      // 降级到localStorage
      return this.fallback.get(key);
    }
  }
  
  async set(key, value) {
    try {
      await this.primary.set(key, value);
    } catch {
      this.fallback.set(key, value);
    }
  }
  
  async exportToFile() {
    const data = await this.primary.getAll();
    return this.file.save(data);
  }
  
  async importFromFile(fileHandle) {
    const data = await this.file.load(fileHandle);
    await this.primary.setAll(data);
  }
}
```

```javascript
// js/components/search.js
class TaskSearch {
  constructor(container, onResults) {
    this.container = container;
    this.onResults = onResults;
    this.history = this.loadHistory();
    
    this.render();
    this.bindEvents();
  }
  
  render() {
    this.container.innerHTML = `
      <div class="search-wrapper">
        <div class="search-input-group">
          <svg class="search-icon" width="16" height="16" viewBox="0 0 24 24">
            <circle cx="11" cy="11" r="8" fill="none" stroke="currentColor" stroke-width="2"/>
            <path d="m21 21-4.35-4.35" fill="none" stroke="currentColor" stroke-width="2"/>
          </svg>
          <input type="text" class="search-input" placeholder="搜索任务..." />
          <button class="search-clear" style="display:none">×</button>
        </div>
        <div class="search-results"></div>
      </div>
    `;
    
    this.input = this.container.querySelector('.search-input');
    this.clearBtn = this.container.querySelector('.search-clear');
    this.resultsEl = this.container.querySelector('.search-results');
  }
  
  bindEvents() {
    // 实时搜索（防抖）
    this.input.addEventListener('input', 
      debounce(() => this.search(this.input.value), 200)
    );
    
    // 清除按钮
    this.clearBtn.addEventListener('click', () => {
      this.input.value = '';
      this.clear();
      this.input.focus();
    });
    
    // ESC关闭
    this.input.addEventListener('keydown', (e) => {
      if (e.key === 'Escape') this.clear();
    });
  }
  
  search(keyword) {
    if (!keyword.trim()) {
      this.clear();
      return;
    }
    
    const { tasks } = store.getState();
    const lowerKeyword = keyword.toLowerCase();
    
    const results = tasks.filter(task => 
      task.title.toLowerCase().includes(lowerKeyword) ||
      (task.project && task.project.toLowerCase().includes(lowerKeyword))
    ).slice(0, 20); // 限制结果数量
    
    this.renderResults(results, keyword);
    this.saveHistory(keyword);
    this.onResults(results);
  }
  
  renderResults(results, keyword) {
    if (results.length === 0) {
      this.resultsEl.innerHTML = '<div class="search-empty">未找到匹配任务</div>';
      return;
    }
    
    this.resultsEl.innerHTML = results.map(task => `
      <div class="search-result-item" data-id="${task.id}">
        <div class="search-result-title">${this.highlight(task.title, keyword)}</div>
        <div class="search-result-meta">
          ${task.project ? `<span class="search-result-project">${task.project}</span>` : ''}
          <span class="search-result-date">${task.date}</span>
        </div>
      </div>
    `).join('');
  }
  
  highlight(text, keyword) {
    const regex = new RegExp(`(${keyword})`, 'gi');
    return text.replace(regex, '<mark>$1</mark>');
  }
  
  clear() {
    this.resultsEl.innerHTML = '';
    this.clearBtn.style.display = 'none';
    this.onResults(null);
  }
  
  loadHistory() {
    try {
      return JSON.parse(localStorage.getItem('search_history')) || [];
    } catch {
      return [];
    }
  }
  
  saveHistory(keyword) {
    this.history = [keyword, ...this.history.filter(k => k !== keyword)].slice(0, 10);
    localStorage.setItem('search_history', JSON.stringify(this.history));
  }
}
```

```javascript
// js/interactions/dragTask.js
class TaskDragger {
  constructor() {
    this.dragging = false;
    this.dragTask = null;
    this.dragOffset = { x: 0, y: 0 };
    this.ghost = null;
    this.originalDate = null;
    this.originalStartMin = null;
  }
  
  bind(taskElement, task) {
    taskElement.addEventListener('mousedown', (e) => {
      if (e.target.closest('.task-checkbox')) return;
      this.startDrag(e, taskElement, task);
    });
  }
  
  startDrag(e, element, task) {
    this.dragging = true;
    this.dragTask = task;
    this.originalDate = task.date;
    this.originalStartMin = task.startMin;
    
    const rect = element.getBoundingClientRect();
    this.dragOffset = {
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    };
    
    // 创建拖拽幽灵
    this.ghost = element.cloneNode(true);
    this.ghost.className += ' drag-ghost';
    this.ghost.style.width = rect.width + 'px';
    this.ghost.style.position = 'fixed';
    this.ghost.style.pointerEvents = 'none';
    this.ghost.style.opacity = '0.8';
    this.ghost.style.zIndex = '1000';
    document.body.appendChild(this.ghost);
    
    // 原元素半透明
    element.style.opacity = '0.3';
    this.originalElement = element;
    
    this.updateGhostPosition(e);
    
    document.addEventListener('mousemove', this.onMouseMove);
    document.addEventListener('mouseup', this.onMouseUp);
    
    document.body.classList.add('dragging');
  }
  
  onMouseMove = (e) => {
    if (!this.dragging) return;
    this.updateGhostPosition(e);
    this.updateDropTarget(e);
  }
  
  onMouseUp = (e) => {
    if (!this.dragging) return;
    
    this.dragging = false;
    document.body.classList.remove('dragging');
    
    // 清理
    if (this.ghost) {
      this.ghost.remove();
      this.ghost = null;
    }
    
    if (this.originalElement) {
      this.originalElement.style.opacity = '';
    }
    
    document.removeEventListener('mousemove', this.onMouseMove);
    document.removeEventListener('mouseup', this.onMouseUp);
    
    // 应用变更
    this.applyDrop(e);
  }
  
  updateGhostPosition(e) {
    if (!this.ghost) return;
    this.ghost.style.left = (e.clientX - this.dragOffset.x) + 'px';
    this.ghost.style.top = (e.clientY - this.dragOffset.y) + 'px';
  }
  
  updateDropTarget(e) {
    // 查找目标日期列
    const columns = document.querySelectorAll('.day-column');
    let targetColumn = null;
    
    columns.forEach(col => {
      const rect = col.getBoundingClientRect();
      if (e.clientX >= rect.left && e.clientX <= rect.right) {
        targetColumn = col;
      }
    });
    
    // 高亮目标列
    columns.forEach(col => col.classList.remove('drop-target'));
    if (targetColumn) {
      targetColumn.classList.add('drop-target');
    }
  }
  
  async applyDrop(e) {
    const columns = document.querySelectorAll('.day-column');
    let targetColumn = null;
    
    columns.forEach(col => {
      const rect = col.getBoundingClientRect();
      if (e.clientX >= rect.left && e.clientX <= rect.right) {
        targetColumn = col;
      }
    });
    
    if (!targetColumn) return;
    
    const rect = targetColumn.getBoundingClientRect();
    const y = e.clientY - rect.top - this.dragOffset.y;
    const newStartMin = pxToM(Math.max(0, y));
    const duration = this.dragTask.endMin - this.dragTask.startMin;
    const newEndMin = Math.min(newStartMin + duration, 24 * 60);
    
    const newDate = targetColumn.dataset.date;
    
    // 应用变更
    await taskStore.update(this.dragTask.id, {
      date: newDate,
      startMin: newStartMin,
      endMin: newEndMin
    });
    
    render();
  }
}

// 初始化
const taskDragger = new TaskDragger();

// 在渲染任务时绑定
function createTaskElement(task) {
  const el = document.createElement('div');
  // ... 创建元素
  
  taskDragger.bind(el, task);
  
  return el;
}
```

```javascript
// 生成重复任务
function generateRecurringTasks(baseTask, startDate, endDate) {
  const tasks = [];
  let currentDate = new Date(startDate);
  
  while (currentDate <= endDate) {
    // 检查是否在排除日期
    const dateStr = ds(currentDate);
    if (!baseTask.recurrence.exceptions.includes(dateStr)) {
      tasks.push({
        ...baseTask,
        id: uid(),
        date: dateStr,
        parentId: baseTask.id
      });
    }
    
    // 计算下一个日期
    switch (baseTask.recurrence.rule) {
      case RECURRENCE_RULES.DAILY:
        currentDate.setDate(currentDate.getDate() + baseTask.recurrence.interval);
        break;
      case RECURRENCE_RULES.WEEKLY:
        currentDate.setDate(currentDate.getDate() + 7 * baseTask.recurrence.interval);
        break;
      case RECURRENCE_RULES.MONTHLY:
        currentDate.setMonth(currentDate.getMonth() + baseTask.recurrence.interval);
        break;
      // ... 其他规则
    }
  }
  
  return tasks;
}
```

```javascript
        cat.name
      ].map(cell => `"${cell}"`).join(',');
    });
    
    const csv = [headers.join(','), ...rows].join('\n');
    const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8' });
    this.download(blob, `日程导出_${ds(new Date())}.csv`);
  }
  
  // 导入CSV
  async importCSV(file) {
    const text = await file.text();
    const lines = text.split('\n').filter(line => line.trim());
    
    // 跳过标题行
    const dataLines = lines.slice(1);
    
    const importedTasks = dataLines.map(line => {
      const [date, startTime, endTime, title, project, categoryName] = 
        line.split(',').map(cell => cell.replace(/"/g, '').trim());
      
      const category = categories.find(c => c.name === categoryName) || categories[0];
      
      return {
        id: uid(),
        date: date,
        startMin: this.timeToMinutes(startTime),
        endMin: this.timeToMinutes(endTime),
        title: title,
        project: project || '',
        categoryId: category.id
      };
    });
    
    tasks = [...tasks, ...importedTasks];
    saveTasks();
    render();
    
    alert(`成功导入 ${importedTasks.length} 个任务`);
  }
  
  download(blob, filename) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    a.click();
    URL.revokeObjectURL(url);
  }
  
  validateImportData(data) {
    return data && Array.isArray(data.tasks) && Array.isArray(data.categories);
  }
  
  mergeData(importData) {
    // 合并分类
    const existingCatIds = new Set(categories.map(c => c.id));
    const newCategories = importData.categories.filter(c => !existingCatIds.has(c.id));
    
    // 合并项目
    const existingProjNames = new Set(projects.map(p => p.name));
    const newProjects = importData.projects.filter(p => !existingProjNames.has(p.name));
    
    // 合并任务（避免重复）
    const existingTaskIds = new Set(tasks.map(t => t.id));
    const newTasks = importData.tasks.filter(t => !existingTaskIds.has(t.id));
    
    return {
      tasks: [...tasks, ...newTasks],
      categories: [...categories, ...newCategories],
      projects: [...projects, ...newProjects]
    };
  }
  
  timeToMinutes(timeStr) {
    const [hours, minutes] = timeStr.split(':').map(Number);
    return hours * 60 + minutes;
  }
}
```

## 5. 技术选型对比

### 5.1 前端框架选型

| 方案 | 优点 | 缺点 | 推荐度 |
|------|------|------|--------|
| **继续原生** | 无依赖、加载快、学习成本低 | 开发效率低、代码难维护 | ⭐⭐⭐ |
| **Vue 3** | 学习曲线平缓、生态丰富、双向绑定 | 需要构建工具 | ⭐⭐⭐⭐⭐ |
| **React** | 组件化强、生态最丰富 | 学习曲线陡、JSX不够直观 | ⭐⭐⭐⭐ |
| **Svelte** | 编译时框架、性能极佳、代码简洁 | 生态较小、社区不成熟 | ⭐⭐⭐⭐ |

**建议：** 如果项目需要长期维护和功能扩展，建议迁移到 **Vue 3**。如果只是个人使用、功能简单，继续原生即可。

### 5.2 图表库选型

| 库 | 大小 | 特点 | 推荐度 |
|---|------|------|--------|
| **Chart.js** | 60KB | 简单易用、文档完善 | ⭐⭐⭐⭐⭐ |
| **ECharts** | 300KB | 功能强大、中文友好 | ⭐⭐⭐⭐ |
| **D3.js** | 100KB | 极度灵活、学习成本高 | ⭐⭐⭐ |
| **原生Canvas** | 0 | 无依赖、完全控制 | ⭐⭐ |

**建议：** 优先选择 **Chart.js**，轻量且满足需求。

### 5.3 存储方案选型

| 方案 | 容量 | 性能 | 兼容性 | 推荐度 |
|------|------|------|--------|--------|
| **localStorage** | 5MB | 同步、快 | 100% | ⭐⭐⭐ |
| **IndexedDB** | 无限 | 异步、可索引 | 98% | ⭐⭐⭐⭐⭐ |
| **File System API** | 无限 | 直接文件操作 | 70% | ⭐⭐⭐ |
| **OPFS** | 无限 | 高性能 | 80% | ⭐⭐⭐⭐ |

**建议：** 主存储使用 **IndexedDB**，导出功能保留 **File System API**。
